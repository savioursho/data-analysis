Lazy quick start

# 実行されない
(pl.read_csv("https://j.mp/iriscsv")
    .lazy()
    .filter(pl.col("sepal_length") > 5)
    .groupby("species", maintain_order=True)
    .agg(pl.all().sum())
)
naive plan: (run LazyFrame.describe_optimized_plan() to see the optimized plan)

# .collect()で実行される

    .agg(pl.all().sum())

)
